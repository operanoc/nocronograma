  1458→function parseXLSXWorkbook(wb){
  1459→  var result={pollings:[],backups:[],processes:[]};
  1460→  // Try to find sheets by name (match export format or common names)
  1461→  var sheetNames=wb.SheetNames;
  1462→  var pollSheet=null,bkpSheet=null,procSheet=null;
  1463→  sheetNames.forEach(function(n){
  1464→    var u=n.toUpperCase();
  1465→    if(u.indexOf('POLLING')>=0)pollSheet=n;
  1466→    else if(u.indexOf('BACKUP')>=0||u.indexOf('BKP')>=0)bkpSheet=n;
  1467→    else if(u.indexOf('PROCESO')>=0)procSheet=n;
  1468→  });
  1469→  // If no named sheets found, try by column headers
  1470→  if(!pollSheet&&!bkpSheet&&!procSheet){
  1471→    sheetNames.forEach(function(n){
  1472→      var ws=wb.Sheets[n];
  1473→      var json=XLSX.utils.sheet_to_json(ws,{header:1});
  1474→      if(json.length<2)return;
  1475→      var hdr=json[0].map(function(h){return String(h||'').toUpperCase();});
  1476→      if(hdr.indexOf('SERVIDOR')>=0&&hdr.indexOf('FASE')>=0)pollSheet=n;
  1477→      else if(hdr.indexOf('NOMBRE')>=0&&(hdr.indexOf('HORA INICIO')>=0||hdr.indexOf('ESTADO')>=0)){
  1478→        if(hdr.indexOf('JOB')>=0)bkpSheet=n;
  1479→        else procSheet=n;
  1480→      }
  1481→    });
  1482→  }
  1483→  // Parse Pollings sheet
  1484→  if(pollSheet){
  1485→    var ws=wb.Sheets[pollSheet];
  1486→    var rows=XLSX.utils.sheet_to_json(ws,{header:1});
  1487→    for(var i=0;i<rows.length;i++){
  1488→      var r=rows[i];
  1489→      if(!r||!r[0])continue;
  1490→      var srv=String(r[0]||'').trim();
  1491→      // Skip header rows
  1492→      if(srv.toUpperCase()==='SERVIDOR'||srv.toUpperCase()==='POLLINGS')continue;
  1493→      var region=String(r[1]||'').trim();
  1494→      var phase=String(r[2]||'').trim().toUpperCase();
  1495→      var time=String(r[3]||'').trim();
  1496→      if(!srv)continue;
  1497→      if(phase!=='BEGINNING'&&phase!=='ENDING')continue;
  1498→      result.pollings.push({server:srv,region:region||getRegion(srv),phase:phase,time:time||''});
  1499→    }
  1500→  }
  1501→  // Parse Backups sheet
  1502→  if(bkpSheet){
  1503→    var ws2=wb.Sheets[bkpSheet];
  1504→    var rows2=XLSX.utils.sheet_to_json(ws2,{header:1});
  1505→    for(var j=0;j<rows2.length;j++){
  1506→      var r2=rows2[j];
  1507→      if(!r2||!r2[0])continue;
  1508→      var bname=String(r2[0]||'').trim();
  1509→      if(bname.toUpperCase()==='NOMBRE'||bname.toUpperCase()==='BACKUPS')continue;
  1510→      result.backups.push({
  1511→        name:bname,
  1512→        iniTime:String(r2[1]||'').trim(),
  1513→        endTime:String(r2[2]||'').trim(),
  1514→        duration:String(r2[3]||'').trim(),
  1515→        status:String(r2[4]||'').trim(),
  1516→        job:String(r2[5]||'').trim()
  1517→      });
  1518→    }
  1519→  }
  1520→  // Parse Processes sheet
  1521→  if(procSheet){
  1522→    var ws3=wb.Sheets[procSheet];
  1523→    var rows3=XLSX.utils.sheet_to_json(ws3,{header:1});
  1524→    for(var k=0;k<rows3.length;k++){
  1525→      var r3=rows3[k];
  1526→      if(!r3||!r3[0])continue;
  1527→      var pname=String(r3[0]||'').trim();
  1528→      if(pname.toUpperCase()==='NOMBRE'||pname.toUpperCase()==='PROCESOS')continue;
  1529→      result.processes.push({
  1530→        name:pname,
  1531→        status:String(r3[1]||'').trim()
  1532→      });
